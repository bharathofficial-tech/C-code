#include <iostream>
using namespace std;

int main() {
    int num;
    cin >> num;  // Input number
    int sum = 0;
    for (int i = 1; i <= num / 2; ++i) {
        if (num % i == 0) {
            sum += i;
        }
    }
    if (sum > num) {
        cout << "Abundant Number" << endl;
    } else {
        cout << "Not Abundant Number" << endl;
    }

    return 0;
}
